import os
from PIL import Image

def toplu_boyutlandir_recursive(input_root, output_root, target_size=(563, 563)):
    # Desteklenen dosya uzantıları
    valid_extensions = (".png", ".PNG")

    # Tüm alt dizinleri ve dosyaları tara
    for root, dirs, files in os.walk(input_root):
        for filename in files:
            if filename.endswith(valid_extensions):
                # Mevcut dosyanın tam yolu
                img_path = os.path.join(root, filename)

                # Hedef klasör yapısını belirle
                # Giriş dizini ile dosya arasındaki göreceli yolu alır (örn: goz1/alt/bekle)
                relative_path = os.path.relpath(root, input_root)
                target_dir = os.path.join(output_root, relative_path)

                # Hedef alt klasör yoksa oluştur
                if not os.path.exists(target_dir):
                    os.makedirs(target_dir)

                # İşleme başla
                try:
                    with Image.open(img_path) as img:
                        old_width, old_height = img.size

                        # Yeni şeffaf tuval (RGBA)
                        new_img = Image.new("RGBA", target_size, (0, 0, 0, 0))

                        # Ortalamak için koordinat hesapla
                        x_offset = (target_size[0] - old_width) // 2
                        y_offset = (target_size[1] - old_height) // 2

                        # Orijinal görseli yapıştır
                        new_img.paste(img, (x_offset, y_offset))

                        # Hedef yolu oluştur ve kaydet
                        save_path = os.path.join(target_dir, filename)
                        new_img.save(save_path)
                        print(f"Başarılı: {save_path}")

                except Exception as e:
                    print(f"Hata oluştu ({img_path}): {e}")

# Klasör yolları
giris_dizini = "orijinal_resimler"
cikis_dizini = "kare_resimler"

from PIL import Image
import os

# Ayarlar
input_files = [f"goz/goz1/alt/sola-don/alt-sola-don-{i}.png" for i in range(12)]
angles = [0, 10, 20, 30, 40, 45, -45, -40, -30, -20, -10, 0]

def process_image(filename, angle):
    if not os.path.exists(filename):
        print(f"Dosya bulunamadı: {filename}")
        return

    # Görseli aç (RGBA modunda olduğundan emin ol)
    with Image.open(filename).convert("RGBA") as img:
        orig_w, orig_h = img.size

        # 1. İçeriğin sınırlarını (bounding box) bul
        # Bu işlem şeffaf olmayan en dış pikselleri tespit eder
        bbox = img.getbbox()
        if not bbox:
            print(f"Hata: {filename} içeriği boş.")
            return

        # 2. Sadece içeriği kırp
        #content = img.crop(bbox)
        content=img

        # 3. Kırpılan içeriği döndür
        # expand=True içeriğin dönme esnasında kesilmemesini sağlar
        rotated_content = content.rotate(angle, resample=Image.BICUBIC, expand=True)

        # 4. Orijinal boyutta yeni, tamamen şeffaf bir tuval oluştur
        new_canvas = Image.new("RGBA", (orig_w, orig_h), (0, 0, 0, 0))

        # 5. Döndürülmüş içeriğin merkezini, tuvalin merkezine hesapla
        rot_w, rot_h = rotated_content.size

        # Yapıştırma koordinatları (Merkezi hizalama)
        paste_x = (orig_w - rot_w) // 2
        paste_y = (orig_h - rot_h) // 2

        # 6. İçeriği yeni tuvale yapıştır
        new_canvas.paste(rotated_content, (paste_x, paste_y), rotated_content)

        # Sonucu kaydet
        output_name = f"centered_{filename}"
        new_canvas.save(output_name)
        print(f"İşlendi: {filename} -> {output_name} ({angle} derece, ortalandı)")

# Tüm görselleri döngü ile işle
for i in range(len(input_files)):
    process_image(input_files[i], angles[i])

print("\nİşlem tamamlandı. Tüm görseller içerik bazlı ortalandı.")
